# print不换行

int()
float()
str()

# 日期

# 键盘输入

# 随机数

# 数据集：可自动获取指定的数字数组

range(start, end, step)     # step 是数间间隔，比如 2 4 6 8这种，本质是有序类型的切片操作

# 字符串相关
"分隔符".join(列表名)   # 列表的字符串拼接
",".join(["Jack", "Tom", "Machel"])

str.match("正则式", "字符串") # 正则表达式匹配字符串，返回布尔值
